### 확실성 계산 시스템

```typescript
interface 확실성분석 {
  긍정신호: {
    항목: string;
    점수: number;
  }[];
  부정신호: {
    항목: string;
    점수: number;
  }[];
  총점: number;
  권장전략: '공격적' | '신중' | '방어적';
}

function 확실성계산(신호들: 신호[]): 확실성분석 {
  const 긍정점수 = 신호들
    .filter(s => s.유형 === '긍정')
    .reduce((sum, s) => sum + s.점수, 0);
    
  const 부정점수 = 신호들
    .filter(s => s.유형 === '부정')
    .reduce((sum, s) => sum + s.점수, 0);
    
  const 총점 = 긍정점수 - 부정점수;
  
  let 권장전략: string;
  if (총점 >= 80) 권장전략 = '공격적';
  else if (총점 >= 50) 권장전략 = '신중';
  else 권장전략 = '방어적';
  
  return {
    긍정신호: /* ... */,
    부정신호: /* ... */,
    총점,
    권장전략
  };
}
```
